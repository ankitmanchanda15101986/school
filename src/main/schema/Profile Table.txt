CREATE TABLE STUDENT_PROFILE(
   ENROLLMENT_ID INT NOT NULL UNIQUE, 
   USER_NAME VARCHAR (50) NOT NULL,
   ROLL_NUMBER INT NOT NULL,
   FIRST_NAME VARCHAR (50) NOT NULL,
   LAST_NAME VARCHAR (50) NOT NULL,
   STUDENT_DOB DATE NOT NULL,
   AADHAR_NUMBER VARCHAR (20) NOT NULL,
   MOTHERS_NAME VARCHAR (50) NOT NULL,   
   MOTHERS_DOB DATE  NOT NULL,  
   MOTHERS_AADHAR_NUMBER VARCHAR (50) NOT NULL,  
   MOTHERS_PAN_NUMBER VARCHAR (50) NOT NULL,  
   FATHERS_NAME VARCHAR (50) NOT NULL,  
   FATHERS_DOB DATE  NOT NULL,  
   FATHERS_AADHAR_NUMBER VARCHAR (50) NOT NULL,  
   FATHERS_PAN_NUMBER VARCHAR (50) NOT NULL,  
   MOTHER_CONTACT INT (20) NOT NULL,
   FATHER_CONTACT INT (20) NOT NULL,   
   EMAIL_ID VARCHAR (50) NOT NULL,  
   ADDRESS_LINE1 VARCHAR (50) NOT NULL,  
   ADDRESS_LINE2 VARCHAR (50),  
   STATE VARCHAR (50) NOT NULL, 
   PINCODE VARCHAR (50), 
   COUNTRY VARCHAR (50) NOT NULL,
   PRIMARY KEY (ENROLLMENT_ID),
   FOREIGN KEY (USER_NAME) REFERENCES PARENT_LOGIN(USER_NAME)
);

CREATE TABLE PARENT_LOGIN(
   USER_NAME VARCHAR (50) NOT NULL UNIQUE, 
   PASSWORD VARCHAR (50) NOT NULL,
   OTP INT ,
   SECRET_QUESTION1 VARCHAR (50) NOT NULL ,
   ANSWER1 VARCHAR (50) NOT NULL ,
   SECRET_QUESTION2 VARCHAR (50) NOT NULL ,
   ANSWER2 VARCHAR (50) NOT NULL ,
   SECRET_QUESTION3 VARCHAR (50) NOT NULL ,
   ANSWER3 VARCHAR (50) NOT NULL ,
   OLD_PASSWORD1 VARCHAR (50) NOT NULL ,
   OLD_PASSWORD2 VARCHAR (50) NOT NULL ,
   OLD_PASSWORD3 VARCHAR (50) NOT NULL ,
   PRIMARY KEY (USER_NAME)
);
